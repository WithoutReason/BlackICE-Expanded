USA_save_the_aussies = {
	allowed = { tag = USA }
	enable = {
		democratic_govt = yes
		is_in_faction = no
		has_war = no

		has_opinion = {
			target = AST
			value > 0 # not if we hate AST
		}
		AST = {
			has_war_with = JAP
			democratic_govt = yes		
			surrender_progress > 0.15
		}
	}
	abort = {
		OR = {
			has_war_with = AST
			is_in_faction_with = JAP
			JAP = {	
				NOT = { has_war_with = AST } 
			}
		}
	}
	
	ai_strategy = {
		type = contain
		id = "JAP"			
		value = 200
	}
	
	ai_strategy = {
		type = protect
		id = "AST"			
		value = 200
	}
}
